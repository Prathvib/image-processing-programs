#logical operations
import cv2
import numpy as np
import matplotlib.pyplot as plt
img1=cv2.imread('1.png')
img2=cv2.imread('image1.png')

bw_and=cv2.bitwise_and(img1,img2)
bw_or=cv2.bitwise_or(img1,img2)
bw_not=cv2.bitwise_not(img1)

plt.imshow(bw_and)
plt.show()

plt.show()
